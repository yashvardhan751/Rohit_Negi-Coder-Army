#include <iostream>
using namespace std;
int main() 
{
    int row,col;
    int k=1;
    for(row=1;row<=5;row+=1){
        for(col=1;col<=5;col++){
            cout<<k<<" ";
            k+=1;
        }
        
        cout<<endl;
    }
}
